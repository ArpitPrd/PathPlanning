\section*{Optimization Model}

\subsection*{Constraints}

\begin{align}
& \text{(Max Battery Limit)} && b_n^k \leq b_{\text{full}} \\
& \text{(Min Battery Limit)} && b_n^k \geq e_{\text{base}} \\
& \text{(Turn vs Move)} && y_{\text{turn},n}^k + y_{\text{mov},n}^k \leq 1 \\
& \text{(Exchange vs No-Exchange)} && Y_{\text{exchange},n}^k + Y_{\text{noexchange},n}^k \leq 1 \\
& \text{(Binary Turn Limit)} && y_{\text{turn},n}^k \geq 0 \\
& \text{(Binary Move Limit)} && y_{\text{mov},n}^k \geq 0 \\
& \text{(Binary Exchange Limit)} && Y_{\text{exchange},n}^k \geq 0 \\
& \text{(Binary No-Exchange Limit)} && Y_{\text{noexchange},n}^k \geq 0 \\
& \text{(Exchange implies station presence)} && Y_{\text{exchange},n}^k \leq z_{\text{base\_station},n}^k
\end{align}

\subsection*{Battery Update Rule}

\begin{align}
b_n^{k+1} = & \; Y_{\text{noexchange},n}^k \cdot \Big( b_n^k - y_{\text{turn},n}^k \cdot b_{\text{turn}} - y_{\text{mov},n}^k \cdot b_{\text{mov}} \Big) \nonumber \\
& + Y_{\text{exchange},n}^k \cdot b_{\text{full}}
\end{align}

\subsection*{Objective Function (Multi-Objective)}

\begin{align}
\max \quad & \sum_{i=1}^{M} c_i \; , \\
\min \quad & \sum_{n,k} \Big( y_{\text{turn},n}^k \cdot b_{\text{turn}} + y_{\text{mov},n}^k \cdot b_{\text{mov}} + Y_{\text{exchange},n}^k \cdot \big( b_{\text{full}} - b_n^k \big) \Big)
\end{align}
